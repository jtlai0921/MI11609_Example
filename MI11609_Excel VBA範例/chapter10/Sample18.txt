'初始化
Sub resetSheet()
	'編號的更新
	Call getNewID
	With Worksheets("輸入")
		.Range("F2").Value = Date '發售日
		.Range("B4").Value = "" '負責人
		.Range("C11").Value = "" '案件名稱
		.Range("B14:C23,E14:E23").ClearContents '清除輸入內容
		'單價自動輸入
		.Range("D14:D23").Formula = _
			"=IFERROR(VLOOKUP(B14,清單!$F$4:$G$8,2,FALSE),"""")"
		.Range("F14:F23").FormulaR1C1 = _
			"=IFERROR(R[0]C[-2]*R[0]C[-1],"""")"
	End With
End Sub
